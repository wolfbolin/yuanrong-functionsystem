project(http)

message(STATUS "Begin to configure http")
add_library(http STATIC "")
get_property(UTILS_INCLUDE GLOBAL PROPERTY "UTILS_INCLUDE")

include_directories(${UTILS_INCLUDE})
set_target_properties(http PROPERTIES UNITY_BUILD ON)
aux_source_directory(${CMAKE_CURRENT_LIST_DIR} HTTP_SRCS)
add_dependencies(http status)
target_sources(http PRIVATE ${HTTP_SRCS})
target_link_libraries(http PUBLIC status)
